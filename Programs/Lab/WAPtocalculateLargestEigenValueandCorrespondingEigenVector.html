#include <iostream>
#include <cmath>
#define e 0.0001
using namespace std;

int main()
{
	system("CLS");
	float x[100], y[100], c[100][100], eigenvalue = 0, neweigenvalue, error, sum;
	int i, j, n;

	cout << "Enter the dimension of Square Matrix: ";
	cin >> n;

	cout << "Enter the Matrix:" << endl;
	for (i = 1; i <= n; i++)
	{
		for (j = 1; j <= n; j++)
			cin >> c[i][j];
	}

	cout << "Enter the Initial Eigen Vector: " << endl;
	for (i = 1; i <= n; i++)
		cin >> x[i];

top:
	for (i = 1; i <= n; i++)
	{
		sum = 0;
		for (j = 1; j <= n; j++)
		{
			sum = sum + c[i][j] * x[j];
		}
		y[i] = sum;
	}

	neweigenvalue = fabs(y[1]);
	for (i = 2; i <= n; i++)
	{
		if (fabs(y[i]) > neweigenvalue)
			neweigenvalue = fabs(y[i]);
	}

	for (i = 1; i <= n; i++)
		x[i] = y[i] / neweigenvalue;

	error = fabs(eigenvalue - neweigenvalue);
	eigenvalue = neweigenvalue;
	if (error > e)
		goto top;

	cout << "The Eigen Value is: " << eigenvalue << endl;
	cout << "The Eigen Vector is: " << endl;

	for (i = 1; i <= n; i++)
		cout << x[i] << endl;
	return 0;
}
