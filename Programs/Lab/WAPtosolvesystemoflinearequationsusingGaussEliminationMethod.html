// WAP to solve system of linear equations using Gauss-Elimination Method.

#include <iostream>
using namespace std;

int main()
{
	system("CLS");
	int n, i, j, k;
	float c[100][100], fact, x[100], sum;

	cout << "Enter the number of Equations: ";
	cin >> n;

	cout << "Enter the coefficients: ";
	for (i = 1; i <= n; i++)
	{
		for (j = 1; j <= n + 1; j++)
		{
			cin >> c[i][j];
		}
		cout << endl
			 << endl;
	}

	for (k = 1; k <= n - 1; k++)
	{
		for (i = k + 1; i <= n; i++)
		{
			fact = c[i][k] / c[k][k];

			for (j = 1; j <= n + 1; j++)
			{
				c[i][j] = c[i][j] - (fact * c[k][j]);
			}
		}
	}

	x[n] = c[n][n + 1] / c[n][n];

	for (i = n - 1; i >= 1; i--)
	{
		sum = 0;
		for (j = i + 1; j <= n; j++)
		{
			sum = sum + c[i][j] * x[j];
		}

		x[i] = (c[i][n + 1] - sum) / c[i][i];
	}

	cout << "The value of X are: " << endl;

	for (i = 1; i <= n; i++)
	{
		cout << "X[" << i << "] = " << x[i] << endl;
	}
	return 0;
}
