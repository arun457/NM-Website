// WAP to calculate the root of non linear equation using Secant method.

#include <iostream>
#include <math.h>
#define f(x) (x * exp(x) - cos(x))
#define e 0.0001
using namespace std;

int main()
{
	system("CLS");

	float x1, x2, x3;

	do
	{
		cout << "Enter the initial guesses: ";
		cin >> x1 >> x2;
	} while (f(x1) - f(x2) == 0);

	while (fabs(x2 - x1) > e)
	{
		x3 = ((x1 * f(x2)) - (x2 * f(x1))) / (f(x2) - f(x1));
		x1 = x2;
		x2 = x3;
	}

	cout << "The root is: " << x3;
	return 0;
}
