// WAP to solve system of linear equations using Gauss-Jordan Method.

#include <iostream>
using namespace std;

int main()
{
	system("CLS");
	int n, i, j, k;
	float c[100][100], fact, pivot;

	cout << "Enter the number of Equations: ";
	cin >> n;

	cout << "Enter the coefficients: ";
	for (i = 1; i <= n; i++)
	{
		for (j = 1; j <= n + 1; j++)
		{
			cin >> c[i][j];
		}
		cout << endl
			 << endl;
	}

	for (k = 1; k <= n; k++)
	{
		pivot = c[k][k];
		for (i = 1; i <= n + 1; i++)
		{
			c[k][i] = c[k][i] / pivot;
		}

		for (i = 1; i <= n; i++)
		{
			if (i == k)
			{
				continue;
			}

			fact = c[i][k];

			for (j = 1; j <= n + 1; j++)
			{
				c[i][j] = c[i][j] - (fact * c[k][j]);
			}
		}
	}

	cout << "The value of X are: " << endl;

	for (i = 1; i <= n; i++)
	{
		cout << "X[" << i << "] = " << c[i][n + 1] << endl;
	}
	return 0;
}
