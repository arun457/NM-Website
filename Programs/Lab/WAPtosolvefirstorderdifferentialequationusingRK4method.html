// WAP to solve first order differential equation using RK - 4 method.

#include <iostream>
using namespace std;

float f(float x, float y)
{
	return ((y * y - x * x) / (y * y + x * x));
}

int main()
{
	system("CLS");
	float m1, m2, m3, m4, m;
	float x0 = 0, y0 = 1, x = 0.4, h = 0.2;

	while (x0 < x)
	{
		m1 = f(x0, y0);
		m2 = f(x0 + h / 2, y0 + m1 * h / 2);
		m3 = f(x0 + h / 2, y0 + m2 * h / 2);
		m4 = f(x0 + h, y0 + m3 * h);

		m = (m1 + 2 * m2 + 2 * m3 + m4) / 6.0;

		x0 = x0 + h;
		y0 = y0 + (m * h);
	}

	cout << "Y = " << y0;
	return 0;
}
