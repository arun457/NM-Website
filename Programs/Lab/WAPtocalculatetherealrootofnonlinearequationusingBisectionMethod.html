// WAP to calculate the real root of non linear equation using Bisection Method.

#include <iostream>
#include <iomanip>
#include <stdlib.h>
#include <math.h>
using namespace std;

#define f(x) (x * exp(x) - cos(x))
#define e 0.0001

float t(float val) // truncationg upto 4 decimal places
{
	return trunc(val * 10000) / 10000;
}

int main()
{
	float x1, x2, x0;
	int N = 1;
	do
	{
		cout << "Enter x1 and x2: ";
		cin >> x1 >> x2;
		
	} while ((f(x1) * f(x2)) > 0);

	cout << setw(4) << "N" << setw(12) << "x1" << setw(12) << "x2" << setw(20) << "f(x1)" << setw(20) << "f(x2)" << setw(20) << "x0" << setw(20) << "f(x0)" << endl;

	while (fabs(x2 - x1) > e)
	{
		x0 = (x1 + x2) / 2;

		cout << setw(4) << N << setw(12) << t(x1) << setw(12) << t(x2) << setw(20) << t(f(x1)) << setw(20) << t(f(x2)) << setw(20) << t(x0) << setw(20) << t(f(x0)) << endl;

		if ((f(x1) * f(x0)) < 0)
		{
			x2 = x0;
		}
		else
		{
			x1 = x0;
		}

		N++;
	}

	cout << "The root is : " << t(x0);
}
